        // --- KIỂM TRA PHẦN CỨNG MICRO / TAI NGHE MÔ PHỎNG ---
        let hwAudioObj = null;
        let hwAudioProgressInterval = null;

        function playHardwareCheckAudio() {
            const btnIcon = document.getElementById('audioCheckIcon');
            const progress = document.getElementById('audioCheckBar');
            
            if (hwAudioObj) {
                window.speechSynthesis.cancel();
                clearInterval(hwAudioProgressInterval);
                hwAudioObj = null;
                btnIcon.className = "fa-solid fa-play";
                progress.style.width = "0%";
                return;
            }

            if ('speechSynthesis' in window) {
                btnIcon.className = "fa-solid fa-pause";
                const checkUtter = new SpeechSynthesisUtterance("Testing your headphones. Can you hear the sound clearly?");
                checkUtter.lang = "en-US";
                checkUtter.onend = () => {
                    btnIcon.className = "fa-solid fa-play";
                    progress.style.width = "100%";
                    clearInterval(hwAudioProgressInterval);
                    hwAudioObj = null;
                };
                
                hwAudioObj = checkUtter;
                window.speechSynthesis.speak(checkUtter);

                let width = 0;
                hwAudioProgressInterval = setInterval(() => {
                    width += 5;
                    if (width <= 100) {
                        progress.style.width = `${width}%`;
                    }
                }, 150);
            } else {
                showSimpleWarning("Lỗi âm thanh", "Trình duyệt của bạn không hỗ trợ công cụ phát tiếng TTS mẫu.");
            }
        }

        function startSkillActiveTest() {

                window.speechSynthesis.cancel();
                clearInterval(hwAudioProgressInterval);
                hwAudioObj = null;
            }

            setAppState('active_test');

            clearInterval(headerTimerObj);
            headerTimerObj = setInterval(() => {
                testSecondsRemaining--;
                if (testSecondsRemaining <= 0) {
                    clearInterval(headerTimerObj);
                    autoSubmitOnTimeOut();
                } else {
                    updateGlobalHeaderTimer();
                }
            }, 1000);

            const testArea = document.getElementById('screen-test-area');
            const speakingArea = document.getElementById('screen-test-speaking');
            
            if (currentSkill === 'speaking') {
                testArea.classList.add('hidden');

                initSpeakingTestSimulator();
            } else {
                testArea.classList.remove('hidden');
                speakingArea.classList.add('hidden');
                loadTwoPanelTestData();
            }
        }

        }

        let listeningAudioPlaying = false;
        let listeningAudioProgressInterval = null;

        function toggleMediaListeningAudio() {
            const btnIcon = document.getElementById('listeningPlayIcon');
            const progress = document.getElementById('listeningMediaProgressBar');
            const timer = document.getElementById('listeningMediaTimer');
            const audioText = tierContentDb[currentTier].listening.audioText;

            if (listeningAudioPlaying) {
                window.speechSynthesis.cancel();
                clearInterval(listeningAudioProgressInterval);
                listeningAudioPlaying = false;
                btnIcon.className = "fa-solid fa-play";
                progress.style.width = "0%";
                timer.innerText = "0:00 / 2:08";
                return;
            }

            if ('speechSynthesis' in window) {
                listeningAudioPlaying = true;
                btnIcon.className = "fa-solid fa-pause";
                
                const listenUtter = new SpeechSynthesisUtterance(audioText);
                listenUtter.lang = "en-US";
                listenUtter.rate = 0.85;
                listenUtter.onend = () => {
                    btnIcon.className = "fa-solid fa-play";
                    progress.style.width = "100%";
                    timer.innerText = "2:08 / 2:08";
                    clearInterval(listeningAudioProgressInterval);
                    listeningAudioPlaying = false;
                };

                window.speechSynthesis.speak(listenUtter);

                let progressVal = 0;
                listeningAudioProgressInterval = setInterval(() => {
                    progressVal += 2;
                    if (progressVal <= 100) {
                        progress.style.width = `${progressVal}%`;
                        const simSeconds = Math.round((progressVal / 100) * 128);
                        const m = Math.floor(simSeconds / 60);
                        const s = simSeconds % 60;
                        timer.innerText = `${m}:${s.toString().padStart(2, '0')} / 2:08`;
                    }
                }, 300);
            } else {
                showSimpleWarning("Lỗi", "Trình duyệt không hỗ trợ phát âm thanh mẫu.");
            }
        }

        function loadSpeakingQuestion(idx) {
            const questionsDb = tierContentDb[currentTier].speaking;
            const currentQ = questionsDb[idx];

            document.getElementById('speakingHeading').innerText = `Question ${idx + 1}`;
            document.getElementById('speakingQuestionTranscript').innerText = `"${currentQ.prompt}"`;

            let listHtml = "";
            questionsDb.forEach((q, qIdx) => {
                const activeClass = qIdx === idx ? "bg-blue-600 text-white font-bold" : "bg-slate-900 text-slate-500";
                listHtml += `<span class="px-3 py-1 rounded text-[10px] shrink-0 ${activeClass}">${(qIdx + 1).toString().padStart(2, '0')}</span>`;

        function startSpeakingState(state) {
            speakingState = state;
            clearInterval(speakingRingInterval);

            const ringCircle = document.getElementById('speakingRingCircle');
            const timerText = document.getElementById('speakingRingTimer');
            const titleText = document.getElementById('speakingStateTitle');
            const descText = document.getElementById('speakingStateDesc');
            const iconEl = document.getElementById('speakingRingIcon');
            const btnStop = document.getElementById('btnSpeakingAction');

            const circumference = 2 * Math.PI * 60; // 376.8

            if (state === 'prep') {
                speakingSeconds = 5;
                titleText.innerText = "Preparation...";
                titleText.className = "text-sm font-extrabold text-orange-400 uppercase tracking-widest";
                descText.innerText = "Chuẩn bị ý kiến của bạn trong 5 giây trước khi tự động thu âm.";
                iconEl.className = "fa-solid fa-clock-rotate-left text-orange-400 text-xl mb-1";
                ringCircle.setAttribute("stroke", "#f97316");
                btnStop.disabled = true;

                if ('speechSynthesis' in window) {
                    window.speechSynthesis.cancel();
                    const promptUtter = new SpeechSynthesisUtterance(tierContentDb[currentTier].speaking[currentSpeakingQuestionIdx].prompt);
                    promptUtter.lang = "en-US";
                    window.speechSynthesis.speak(promptUtter);
                }

                speakingRingInterval = setInterval(() => {
                    speakingSeconds--;
                    timerText.innerText = `00:00:0${speakingSeconds}`;
                    const offset = circumference - (speakingSeconds / 5) * circumference;
                    ringCircle.style.strokeDashoffset = offset;

                    if (speakingSeconds <= 0) {
                        clearInterval(speakingRingInterval);
                        startSpeakingState("recording");
                    }
                }, 1000);
            } 
            else if (state === 'recording') {
                speakingSeconds = 20; // Giới hạn 20s nói thử nghiệm
                titleText.innerText = "Recording...";
                titleText.className = "text-sm font-extrabold text-blue-400 uppercase tracking-widest animate-pulse";
                descText.innerText = "Hệ thống đang thu âm câu trả lời trực tiếp của bạn...";
                iconEl.className = "fa-solid fa-microphone text-blue-500 text-xl mb-1";
                ringCircle.setAttribute("stroke", "#3b82f6");
                btnStop.disabled = false;

                document.getElementById('speakingVideoPulse').classList.remove('hidden');

                speakingRingInterval = setInterval(() => {
                    speakingSeconds--;
                    const printedSec = speakingSeconds.toString().padStart(2, '0');
                    timerText.innerText = `00:00:${printedSec}`;
                    const offset = circumference - (speakingSeconds / 20) * circumference;
                    ringCircle.style.strokeDashoffset = offset;

                    if (speakingSeconds <= 0) {
                        clearInterval(speakingRingInterval);
                        proceedNextSpeakingQuestion();
                    }
                }, 1000);
            }
        }

        function stopSpeakingRecordAction() {
            clearInterval(speakingRingInterval);
            proceedNextSpeakingQuestion();
        }

        function proceedNextSpeakingQuestion() {
            document.getElementById('speakingVideoPulse').classList.add('hidden');
            
            const currentQ = tierContentDb[currentTier].speaking[currentSpeakingQuestionIdx];
            userAnswers.speaking.push({
                prompt: currentQ.prompt,
                transcript: currentQ.transcript
            });

            currentSpeakingQuestionIdx++;
            const totalQs = tierContentDb[currentTier].speaking.length;

            if (currentSpeakingQuestionIdx < totalQs) {
                loadSpeakingQuestion(currentSpeakingQuestionIdx);
            } else {
                showSkillFinishedPopup();
            }
        }

                if (showAnswers && !isCorrect) {

        function updateSidebarTreeStatus(activeSkill) {
            const rdNode = document.getElementById('node-reading');
            const lsNode = document.getElementById('node-listening');
            const spNode = document.getElementById('node-speaking');
            const wrNode = document.getElementById('node-writing');

            [rdNode, lsNode, spNode, wrNode].forEach(node => {
                node.classList.add('opacity-50', 'pointer-events-none');
            });

            let targetNode = null;
            if (activeSkill === 'reading') targetNode = rdNode;
            else if (activeSkill === 'listening') targetNode = lsNode;
            else if (activeSkill === 'speaking') targetNode = spNode;
            else if (activeSkill === 'writing') targetNode = wrNode;

            if (targetNode) {
                targetNode.classList.remove('opacity-50', 'pointer-events-none');
            }

            const readSectionsList = document.getElementById('readingSectionsList');
            if (activeSkill === 'reading') {
                readSectionsList.classList.remove('hidden');
            } else {
                readSectionsList.classList.add('hidden');
            }
        }

        function adjustTextScale(val) {
            textScale = parseInt(val);
            const label = document.getElementById('textScaleLabel');
            const textContainers = [
                document.getElementById('mediaPaneText'),
                document.getElementById('reviewMediaText')
            ];

            const sizes = ["font-size-small", "font-size-medium", "font-size-large"];
            const labels = ["Nhỏ", "Mặc định", "Lớn"];

            label.innerText = labels[textScale - 1];

            textContainers.forEach(container => {
                if (container) {
                    container.className = `flex-1 overflow-y-auto pr-2 text-slate-300 leading-relaxed ${sizes[textScale - 1]}`;
                }
            });
        }
